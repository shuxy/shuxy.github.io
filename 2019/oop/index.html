继承 -> 匿名内部类

type father struct {
	A int
	B int
}

type son struct {
	father
	C int
}

func main() {
	newSon := &son{
		father: father{
			A: 1,
			B: 2,
		},
		C: 3,
	}
	fmt.Println(newSon.A)
}

多态 -> interface

type worker interface {
	Work()
}

type teacher struct {
	Name string
}

type doctor struct {
	Name string
}

func (t *teacher) Work() {
	fmt.Printf("%s is working\n", t.Name)
}

func (d *doctor) Work() {
	fmt.Printf("%s is working\n", d.Name)
}

func main() {
	var baseobject worker
	teacherZ := &teacher{Name:"zhangsan"}
	doctorL := &doctor{Name:"lisi"}
 	// 父类引用指向子类对象
	baseobject = teacherZ
	baseobject.Work()

	baseobject = doctorL
	baseobject.Work()
}
